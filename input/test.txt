fn fizzbuzz(n: i32) {
    for (i: i32 = 1; i <= n; i++) {
        if (i % 15 == 0) {
            println("FizzBuzz");
        } else if (i % 3 == 0) {
            println("Fizz");
        } else if (i % 5 == 0) {
            println("Buzz");
        } else {
            println(i);
        }
    }
}

fn main() -> i32 {
    fizzbuzz(15);
    return 0;
}
